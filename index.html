-- ШАГ 1: Создать таблицу
CREATE TABLE IF NOT EXISTS vacations (
  id               BIGSERIAL PRIMARY KEY,
  employee_name    TEXT          NOT NULL,
  department       TEXT,
  position         TEXT,
  vacation_type    TEXT          NOT NULL DEFAULT 'Ежегодный',
  start_date       DATE          NOT NULL,
  end_date         DATE          NOT NULL,
  days_count       INTEGER       NOT NULL DEFAULT 0,
  status           TEXT          NOT NULL DEFAULT 'На рассмотрении',
  note             TEXT,
  created_at       TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);
-- ШАГ 2: Индексы
CREATE INDEX IF NOT EXISTS idx_vac_start   ON vacations(start_date);
CREATE INDEX IF NOT EXISTS idx_vac_dept    ON vacations(department);
CREATE INDEX IF NOT EXISTS idx_vac_status  ON vacations(status);
CREATE INDEX IF NOT EXISTS idx_vac_created ON vacations(created_at);
-- ШАГ 3: RLS
ALTER TABLE vacations ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow anon read"   ON vacations;
DROP POLICY IF EXISTS "Allow anon insert" ON vacations;
DROP POLICY IF EXISTS "Allow anon update" ON vacations;
DROP POLICY IF EXISTS "Allow anon delete" ON vacations;
CREATE POLICY "Allow anon read"   ON vacations FOR SELECT USING (true);
CREATE POLICY "Allow anon insert" ON vacations FOR INSERT WITH CHECK (true);
CREATE POLICY "Allow anon update" ON vacations FOR UPDATE USING (true);
CREATE POLICY "Allow anon delete" ON vacations FOR DELETE USING (true);
